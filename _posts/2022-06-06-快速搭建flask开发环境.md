   <h4>到这里,一个最简单的flassk程序就可以跑起来了,右键点击在终端中运行python文件</h4>
    ~~~
        # 需要创建一个web应用，FLask中创建应用的对象是flask，引入对象
        from flask import Flask
        
        # 创建对象
        app = Flask(__name__)

        # 路由的地址
        @app.route('/')
        def index():
            # return 代表将数据返回给浏览器
            return "尚学堂-python--1"


        if __name__ == '__main__':
            # 启动web应用
            app.run()
    ~~~
