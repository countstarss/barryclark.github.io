		remove(index){
			if(index < 0 || index > this.size){
				throw new Error("index out of range")
			}
			let current = this.head;

			if(index === 0){
				this.head = current.next
			}else{
				let previous = null;
				let i = 0;
				while(i < index){
					previous = current;
					current = current.next;
					i++;
				}
				previous.next = current.next;
			}
			this.size --;
			return current.value
		}
