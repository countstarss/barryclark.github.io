## 冒泡排序

- 遍历n-1次
- 比较次数递减

### 每一趟把最大值放到最右边

```javascript
function BubbleSort(arr){
    let n = arr.length;
    //遍历趟数
    for (let i = 0; i < n - 1; i++){
        //比较次数
        for (let j = 0; j < n - 1 - i; j++){
            //升序
            if (arr[j] > arr[j + 1]){
                let temp = arr[j+1]
                arr[j+1] = arr[j]
                arr[j] = temp
            }
            // console.log(...arr);
        }
        console.log(...arr,`current is [${i}]`);
    }
}

BubbleSort([21,4,65,9,5,2,125,54,234,87,33])
```
